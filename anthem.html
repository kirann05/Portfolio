Inquiry Classification Pipeline 

class InquiryClassifier:
    def __init__(self, model_name="emilyalsentzer/Bio_ClinicalBERT"):
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForSequenceClassification.from_pretrained(
            model_name, num_labels=8
        )
        self.categories = [
            "Appointment", "Billing", "Clinical", "Prescription",
            "Insurance", "General", "Urgent", "Feedback"
        ]

    def classify(self, inquiry_text):
        """Classifies patient inquiry into 8 distinct categories"""
        inputs = self.tokenizer(
            inquiry_text, return_tensors="pt", 
            truncation=True, max_length=512
        )
        with torch.no_grad():
            outputs = self.model(**inputs)
            probs = torch.softmax(outputs.logits, dim=1)
            confidence, predicted_class = torch.max(probs, dim=1)
        
        return {
            "category": self.categories[predicted_class.item()],
            "confidence": confidence.item(),
            "probabilities": probs.squeeze().tolist()
        }

Fraud Detection Feature Engineering 

-- Feature engineering for fraud detection
WITH claim_features AS (
    SELECT 
        c.claim_id, c.provider_id, c.claim_amount,
        -- Provider historical patterns
        AVG(c2.claim_amount) OVER (
            PARTITION BY c.provider_id 
            ORDER BY c2.claim_date
            ROWS BETWEEN 90 PRECEDING AND 1 PRECEDING
        ) AS provider_avg_amount_90d,
        -- Patient historical patterns
        COUNT(DISTINCT c2.provider_id) OVER (
            PARTITION BY c.patient_id 
            ORDER BY c2.claim_date
            ROWS BETWEEN 90 PRECEDING AND 1 PRECEDING
        ) AS patient_provider_diversity_90d
    FROM claims c
    LEFT JOIN claims c2 ON c.provider_id = c2.provider_id
)
SELECT *,
    CASE 
        WHEN claim_amount > provider_avg_amount_90d * 3 THEN 1 
        ELSE 0 
    END AS amount_anomaly_flag
FROM claim_features
WHERE amount_anomaly_flag = 1;
